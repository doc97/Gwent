1   package fi.riissanen.gwent.game.combat;

3   import java.util.ArrayList;
4   import java.util.List;
5   
6   /**
7    * A class representing a combat row on the game board
8    *
9    * @author Daniel
10   * @see GameBoard
11   */
12  public class CombatRow {
13  
14      private final List<Unit> units = new ArrayList<>();
15  
16      public void addUnit(Unit unit) {
17          if (unit != null) {
18              units.add(unit);
19          }
20      }
21  
22      public int getStrength() {
23          int sum = 0;
24          for (Unit unit : units) {
25              sum += unit.getStrength();
26          }
27          return sum;
28      }
29  
30      public List<Unit> getUnits() {
31          return units;
32      }
33      
34      public int getUnitCount() {
35          return units.size();
36      }
37  }
