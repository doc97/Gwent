28  /**
29   * An asset loader for textures
30   * @see AssetLoader
31   * @author Daniel
32   */
33  public class TextureLoader extends AssetLoader {
34      
35      @Override
36      public Asset load(AssetParams params) {
37          if (!params.getFilename().endsWith(".png")) {
38              throw new IllegalArgumentException("Texture must be a .png");
39          }
40          
41          Object format = params.getAdditionalParams()[0];
42          if (!(format instanceof Format)) {
43              throw new IllegalArgumentException("Texture format is invalid");
44          }
45          
46          this.params = params;
47          ByteBuffer buf;
48          int tWidth;
49          int tHeight;
50          log = "";
51          try (InputStream in = new FileInputStream(params.getFilename())) {
52              PNGDecoder decoder = new PNGDecoder(in);
53              tWidth = decoder.getWidth();
54              tHeight = decoder.getHeight();
55              
56              buf = ByteBuffer.allocateDirect(4 * decoder.getWidth() * decoder.getHeight());
57              decoder.decode(buf, decoder.getWidth() * 4, (Format) format);
58              buf.flip();
59          } catch (IOException e) {
60              log = e.getMessage();
61              return null;
62          }
63  
64          int textureID = glGenTextures();
65          glEnable(GL_TEXTURE_2D);
66          glBindTexture(GL_TEXTURE_2D, textureID);
67          glPixelStorei(GL_UNPACK_ALIGNMENT, 1);
68          glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_MIN_FILTER, GL_LINEAR);
69          glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_MAG_FILTER, GL_LINEAR);
70          glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_WRAP_S, GL_CLAMP_TO_EDGE);
71          glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_WRAP_T, GL_CLAMP_TO_EDGE);
72          glTexImage2D(GL_TEXTURE_2D, 0, GL_RGBA, tWidth, tHeight, 0, GL_RGBA, GL_UNSIGNED_BYTE, buf);
73          glGenerateMipmap(GL_TEXTURE_2D);
74  
75          return new Texture(textureID, tWidth, tHeight);
76      }
77  }
