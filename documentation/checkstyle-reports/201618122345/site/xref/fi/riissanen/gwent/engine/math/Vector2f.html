3   /**
4    * A 2D vector.
5    * @author Daniel
6    */
7   public class Vector2f {
8   
9       public float x;
10      public float y;
11      
12      /**
13       * Creates a vector with components set to 0.
14       */
15      public Vector2f() {
16          x = 0;
17          y = 0;
18      }
19      
20      /**
21       * Creates a vector with x and y components.
22       * @param x The x component
23       * @param y The y component
24       */
25      public Vector2f(float x, float y) {
26          this.x = x;
27          this.y = y;
28      }
29      
30      /**
31       * Sets the vector components.
32       * @param x The x component
33       * @param y The y component
34       */
35      public void set(float x, float y) {
36          this.x = x;
37          this.y = y;
38      }
39      
40      /**
41       * Adds a vector to this one.
42       * @param vec The vector to add
43       */
44      public void add(Vector2f vec) {
45          this.x += vec.x;
46          this.y += vec.y;
47      }
48      
49      /**
50       * Subtracts a vector from this one.
51       * @param vec The vector to subtract
52       */
53      public void sub(Vector2f vec) {
54          this.x -= vec.x;
55          this.y -= vec.y;
56      }
57      
58      /**
59       * Scales this vector with a scalar.
60       * @param scale The scalar
61       */
62      public void scl(float scale) {
63          this.x *= scale;
64          this.y *= scale;
65      }
66      
67      /**
68       * Returns a copy of this vector.
69       * @return The copy of this vector
70       */
71      public Vector2f cpy() {
72          return new Vector2f(x, y);
73      }
74  }
