1   package fi.riissanen.gwent.game.ui;
2   
3   import fi.riissanen.gwent.engine.Engine;
4   import fi.riissanen.gwent.engine.render.SpriteBatch;
5   import fi.riissanen.gwent.engine.render.Texture;
6   import fi.riissanen.gwent.engine.render.Color;
7   import fi.riissanen.gwent.game.Gwent;
8   import fi.riissanen.gwent.game.MatchManager;
9   
10  /**
11   * Graphical representation of health points.
12   * @author Daniel
13   */
14  public class GUIHealth extends GUIComponent {
15  
16      private final GUICrystal[] friendlyCrystals;
17      private final GUICrystal[] enemyCrystals;
18      private final Color red = new Color(1, 0.3f, 0.3f, 1);
19      private final Color white = new Color(1, 1, 1, 1);
20      private final Gwent game;
21      
22      /**
23       * Creates a {@link GUIComponent}.
24       * @param game The game instance
25       * @param crystalTex The texture
26       */
27      public GUIHealth(Gwent game, Texture crystalTex) {
28          this.game = game;
29          friendlyCrystals = new GUICrystal[MatchManager.MAX_LIVES];
30          enemyCrystals = new GUICrystal[MatchManager.MAX_LIVES];
31          
32          int size = 32;
33          for (int i = 0; i < friendlyCrystals.length; i++) {
34              friendlyCrystals[i] = new GUICrystal(crystalTex);
35              friendlyCrystals[i].setSize(size, size);
36              friendlyCrystals[i].setPosition(300 - (friendlyCrystals.length - i) * size,
37                      300);
38              enemyCrystals[i] = new GUICrystal(crystalTex);
39              enemyCrystals[i].setSize(size, size);
40              enemyCrystals[i].setPosition(300 - (friendlyCrystals.length - i) * size,
41                      Engine.INSTANCE.batch.getViewport().getSrcHeight() - 300);
42          }
43      }
44      
45      @Override
46      public void update() {
47          for (int i = 0; i < friendlyCrystals.length; i++) {
48              if (i + 1 <= game.getGameSystem().getMatchManager().getFriendlyLives()) {
49                  friendlyCrystals[i].setColor(red);
50              } else {
51                  friendlyCrystals[i].setColor(white);
52              }
53                  
54              if (i + 1 <= game.getGameSystem().getMatchManager().getEnemyLives()) {
55                  enemyCrystals[i].setColor(red);
56              } else {
57                  enemyCrystals[i].setColor(white);
58              }
59          }
60      }
61      
62      @Override
63      public void render(SpriteBatch batch) {
64          for (int i = 0; i < friendlyCrystals.length; i++) {
65              friendlyCrystals[i].render(batch);
66              enemyCrystals[i].render(batch);
67          }
68      }
69  }
