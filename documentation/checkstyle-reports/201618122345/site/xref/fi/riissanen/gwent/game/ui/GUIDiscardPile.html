1   package fi.riissanen.gwent.game.ui;
2   
3   import fi.riissanen.gwent.engine.render.SpriteBatch;
4   import fi.riissanen.gwent.engine.render.Texture;
5   import fi.riissanen.gwent.game.cards.Card;
6   import java.util.LinkedHashMap;
7   import java.util.Map;
8   
9   /**
10   * Graphical representation of the discard pile.
11   * @author Daniel
12   */
13  public class GUIDiscardPile extends GUIComponent {
14  
15      private final Map<Card, GUICard> cards = new LinkedHashMap<>();
16      private final TextCache cache;
17      private GUICard lastCard;
18      
19      /**
20       * Creates a {@link GUIComponent}.
21       * @param texture The texture
22       * @param cache The text cache
23       */
24      public GUIDiscardPile(Texture texture, TextCache cache) {
25          super(texture);
26          this.cache = cache;
27      }
28  
29      @Override
30      public void render(SpriteBatch batch) {
31          super.render(batch);
32          for (Card card : cards.keySet()) {
33              cards.get(card).render(batch);
34          }
35      }
36      
37      /**
38       * Adds a GUICard, mapped to a Card key.
39       * @param card The key to map the GUI card to 
40       * @param guiCard The GUI component
41       */
42      public void addCard(Card card, GUICard guiCard) {
43          if (!cards.containsKey(card)) {
44              if (cards.size() < 4) {
45                  guiCard.setPosition(x + (cards.size() + 1) * 2,
46                      y + height - GUICard.HEIGHT - (cards.size() + 1) * 3);
47              } else {
48                  guiCard.setPosition(x + 5 * 2,
49                      y + height - GUICard.HEIGHT - 5 * 3);
50              }
51              
52              if (lastCard != null) {
53                  cache.removeText(lastCard.getText());
54              }
55              lastCard = guiCard;
56              cache.addText(guiCard.getText());
57              cards.put(card, guiCard);
58          }
59      }
60      
61      /**
62       * Remove a GUICard based on a Card key.
63       * @param card The key
64       */
65      public void removeCard(Card card) {
66          cards.get(card).destroy();
67          cards.remove(card);
68          
69          int i = 0;
70          for (Card c : cards.keySet()) {
71              GUICard guiCard = cards.get(c);
72              if (i < 4) {
73                  guiCard.setPosition(x + (i + 1) * 2, y - (i + 1) * 3);
74              } else {
75                  guiCard.setPosition(x + 5 * 2, y - 5 * 3);
76              }
77              
78              if (i == cards.size() - 1) {
79                  lastCard = guiCard;
80                  cache.addText(guiCard.getText());
81              }
82              i++;
83          }
84      }
85      
86      /**
87       * Gets the {@link GUICard} linked to a Card.
88       * @param card The card that is linked
89       * @return The GUICard
90       */
91      public GUICard getGUICard(Card card) {
92          return cards.get(card);
93      }
94  }
