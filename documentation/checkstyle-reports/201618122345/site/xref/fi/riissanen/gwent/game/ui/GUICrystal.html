1   package fi.riissanen.gwent.game.ui;
2   
3   import fi.riissanen.gwent.engine.render.Color;
4   import fi.riissanen.gwent.engine.render.SpriteBatch;
5   import fi.riissanen.gwent.engine.render.Texture;
6   
7   /**
8    * A round crystal.
9    * @author Daniel
10   */
11  public class GUICrystal extends GUIComponent {
12  
13      private final Color color = new Color(1, 1, 1, 1);
14  
15      /**
16       * Creates a {@link GUIComponent}.
17       * @param texture The texture
18       */
19      public GUICrystal(Texture texture) {
20          super(texture);
21      }
22      
23      @Override
24      public void render(SpriteBatch batch) {
25          Color oldColor = batch.getColor();
26          batch.setColor(color);
27          super.render(batch);
28          batch.setColor(oldColor);
29      }
30      
31      /**
32       * Sets the color of the crystal.
33       * @param color The color
34       */
35      public void setColor(Color color) {
36          this.color.set(color);
37      }
38      
39      /**
40       * Sets the color components of the crystal.
41       * @param r The red component
42       * @param g The green component
43       * @param b The blue component
44       * @param a The alpha component
45       */
46      public void setColor(float r, float g, float b, float a) {
47          color.set(r, g, b, a);
48      }
49  }
