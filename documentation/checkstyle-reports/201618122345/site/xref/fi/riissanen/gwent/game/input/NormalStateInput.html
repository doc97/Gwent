1   package fi.riissanen.gwent.game.input;

3   import fi.riissanen.gwent.engine.input.InputAdapter;
4   import fi.riissanen.gwent.game.Gwent;
5   import org.lwjgl.glfw.GLFW;
6   
7   /**
8    * Handles all general inputs in the normal state.
9    * @author Daniel
10   */
11  public class NormalStateInput extends InputAdapter {
12  
13      private final Gwent game;
14      
15      /**
16       * Constructor.
17       * @param game The game instance
18       */
19      public NormalStateInput(Gwent game) {
20          this.game = game;
21      }
22      
23      @Override
24      public boolean keyPressed(int key) {
25          if (key == GLFW.GLFW_KEY_SPACE) {
26              game.getGameSystem().passRound();
27          }
28          return false;
29      }
30  }
