3   import fi.riissanen.gwent.engine.Engine;
4   import fi.riissanen.gwent.engine.Logger.LogLevel;
5   import fi.riissanen.gwent.game.cards.neutral.WeatherCard;
6   import java.util.ArrayList;
7   import java.util.Iterator;
8   import java.util.List;

10  /**
11   * A class representing the game board.
12   *
13   * @author Daniel
14   */
15  public class GameBoard {
16  
17      private final CombatRow[] friendlyRows;
18      private final CombatRow[] enemyRows;
19      private final List<WeatherCard> weatherPile;
20      private final List<Unit> savedUnits;
21  
22      /**
23       * Constructor, initializes arrays and lists.
24       */
25      public GameBoard() {
26          friendlyRows = new CombatRow[]{
27              new CombatRow(), new CombatRow(), new CombatRow()
28          };
29          enemyRows = new CombatRow[]{
30              new CombatRow(), new CombatRow(), new CombatRow()
31          };
32          weatherPile = new ArrayList<>();
33          savedUnits = new ArrayList<>();
34      }
35  
36      /**
37       * Adds a unit to a combat row.
38       * @param unit The unit to add
39       * @param type The row to which the unit is to be added
40       * @param isFriendly True if the unit's owner is friendly
41       */
42      public void addUnit(Unit unit, UnitType type, boolean isFriendly) {
43          if (unit == null) {
44              return;
45          }
46          
47          if (!unit.getTypes().contains(type)) {
48              throw new IllegalArgumentException("Unit is not of unit type " + type);
49          }
50          
51          CombatRow[] rows = getRows(isFriendly);
52          int index = type.getIndex();
53          rows[index].addUnit(unit);
54      }
55      
56      /**
57       * Adds a weather card to the weather card pile.
58       * @param weather The card
59       */
60      public void addWeatherCard(WeatherCard weather) {
61          if (weather != null) {
62              weatherPile.add(weather);
63          }
64      }
65      
66      /**
67       * Empties the weather card pile to the players' discard piles.
68       */
69      public void clearWeather() {
70          for (WeatherCard card : weatherPile) {
71              card.getOwner().discardCard(card);
72          }
73          weatherPile.clear();
74      }
75      
76      /**
77       * Empties the combat rows to the players' discard piles.
78       * 
79       * <p>
80       * The units that are saved are not discarded.
81       * @see GameBoard#saveUnit(Unit)
82       */
83      public void clearRows() {
84          for (CombatRow row : getRows(true)) {
85              for (Iterator<Unit> it = row.getUnits().iterator(); it.hasNext();) {
86                  Unit unit = it.next();
87                  if (!savedUnits.contains(unit)) {
88                      unit.getOwner().discardCard(unit.getCard());
89                      it.remove();
90                  }
91              }
92          }
93          savedUnits.clear();
94      }
95      
96      /**
97       * Save a unit from the next combat row clear.
98       * @param u The unit to save
99       */
100     public void saveUnit(Unit u) {
101         savedUnits.add(u);
102     }
103     
104     /**
105      * Prints the state of the game board to the engine log.
106      */
107     public void status() {
108         Engine.INSTANCE.log.write(LogLevel.INFO, "Board state");
109         
110         String rowName = "";
111         int strength = 0;
112         int rowCount = UnitType.values().length;
113         for (int i = 0; i < rowCount * 2; i++) {
114             boolean friendly = (i >= rowCount);
115             UnitType row = UnitType.values()[i % rowCount];
116             rowName = row.toString();
117             strength = getRowStrength(row, friendly);
118             Engine.INSTANCE.log.write(LogLevel.INFO, rowName + ": " + strength);
119             
120             if (i == rowCount - 1) {
121                 Engine.INSTANCE.log.write(LogLevel.INFO, "---");
122             }
123         }
124     }
125 
126     /**
127      * Returns the strength of a row.
128      * @param row The row
129      * @param friendly True if the row belongs to the friendly player
130      * @return The row strength
131      */
132     public int getRowStrength(UnitType row, boolean friendly) {
133         return getRow(friendly, row).getStrength();
134     }
135     
136     /**
137      * Returns the combined strength of all combat rows belonging to a player.
138      * @param friendly True if the rows belong to the friendly player
139      * @return The combined strength
140      */
141     public int getStrength(boolean friendly) {
142         int sum = 0;
143         for (CombatRow row : getRows(friendly)) {
144             sum += row.getStrength();
145         }
146         return sum;
147     }
148     
149     /**
150      * Gets the combined unit count of a player.
151      * @param friendly True if the rows belong to the friendly player
152      * @return The combined unit count
153      */
154     public int getUnitCount(boolean friendly) {
155         int sum = 0;
156         for (CombatRow row : getRows(friendly)) {
157             sum += row.getUnitCount();
158         }
159         return sum;
160     }
161     
162     /**
163      * Gets the combined unit count of both players.
164      * @return The combined unit count
165      */
166     public int getTotalUnitCount() {
167         int friendlyCount = getUnitCount(true);
168         int enemyCount = getUnitCount(false);
169         return friendlyCount + enemyCount;
170     }
171 
172     /**
173      * Gets the size of the weather card pile.
174      * @return The card count
175      */
176     public int getWeatherCardCount() {
177         return weatherPile.size();
178     }
179     
180     /**
181      * Gets the number of units that will be saved the next combat row clear.
182      * @return The unit count
183      */
184     public int getSavedUnitCount() {
185         return savedUnits.size();
186     }
187     
188     /**
189      * Get the rows of a player.
190      * @param friendly True if the rows belong to the friendly player
191      * @return The combat rows
192      */
193     public CombatRow[] getRows(boolean friendly) {
194         return friendly ? friendlyRows : enemyRows;
195     }
196     
197     /**
198      * Gets one row of a player.
199      * @param friendly True if the rows belong to the friendly player
200      * @param row The row type
201      * @return The combat row
202      */
203     public CombatRow getRow(boolean friendly, UnitType row) {
204         return friendly ? friendlyRows[row.getIndex()] : enemyRows[row.getIndex()];
205     }
206     
207     /**
208      * Gets the row that a unit is on.
209      * @param unit The unit to check with
210      * @return The combat row, or null if the unit is not in any combat row
211      */
212     public CombatRow getRow(Unit unit) {
213         for (CombatRow row : getRows(unit.isFriendly())) {
214             if (row.hasUnit(unit)) {
215                 return row;
216             }
217         }
218         return null;
219     }
220     
221     /**
222      * Gets the index of a combat row.
223      * @param row The row to get the index for
224      * @return The index, or -1 if the rows have not been initialised properly
225      */
226     public int getRowIndex(CombatRow row) {
227         CombatRow[] fRows = getRows(true);
228         CombatRow[] eRows = getRows(false);
229         for (int i = 0; i < fRows.length; i++) {
230             if (fRows[i].equals(row) || eRows[i].equals(row)) {
231                 return i;
232             }
233         }
234         return -1;
235     }
