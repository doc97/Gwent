3   import java.io.BufferedReader;
4   import java.io.FileNotFoundException;
5   import java.io.FileReader;
6   import java.io.IOException;
7   import java.util.ArrayList;
8   import java.util.HashMap;
9   import java.util.List;
10  import java.util.Map;
11  
12  /**
13   * Utility class for file handling.
14   * @author Daniel
15   */
16  public class FileUtils {
17      
18      /**
19       * Reads a file as a string.
20       * @param filename The name of the file
21       * @return The content of the file as a String
22       * @throws FileNotFoundException If the file cannot be found
23       * @throws IOException If an I/O error occurs while reading the file
24       */
25      public String readFileAsString(String filename)
26              throws FileNotFoundException, IOException {
27          StringBuilder builder = new StringBuilder();
28          BufferedReader reader = new BufferedReader(new FileReader(filename));
29          String line;
30          while ((line = reader.readLine()) != null) {
31              if (builder.length() > 0) {
32                  builder.append("\n");
33              }
34              builder.append(line);
35          }
36          return builder.toString();
37      }
38      
39      /**
40       * Reads lines of a file into a list.
41       * @param filename The name of the file
42       * @return The lines in a list
43       * @throws FileNotFoundException If the file cannot be found
44       * @throws IOException If there occurs an error while reading the file
45       */
46      public List<String> readLines(String filename)
47              throws FileNotFoundException, IOException {
48          List<String> lines = new ArrayList<>();
49          BufferedReader br = new BufferedReader(new FileReader(filename));
50          String line;
51          while ((line = br.readLine()) != null) {
52              lines.add(line);
53          }
54          return lines;
55      }
56      
57      /**
58       * Parses key-value pairs from a source text.
59       * 
60       * @param src The source text
61       * @param delimeter The delimiter splitting keys from values
62       * @param splitter The delimiter splitting pairs from each other
63       * @return The key-value pairs mapped by the keys
64       */
65      public Map<String, String> getKeyValues(String src, String delimeter,
66              String splitter) {
67          Map<String, String> values = new HashMap<>();
68          
69          for (String part : src.split(splitter)) {
70              String valuePairs[] = part.split(delimeter);
71              if (valuePairs.length == 2) {
72                  values.put(valuePairs[0], valuePairs[1]);
73              }
74          }
75          return values;
76      }
77  }
