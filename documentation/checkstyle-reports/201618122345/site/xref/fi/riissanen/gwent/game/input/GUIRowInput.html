17  /**
18   * GUI input for {@link GUIRowInput}.
19   * @author Daniel
20   */
21  public class GUIRowInput extends GUIInputAdapter {
22  
23      private final Gwent game;
24      
25      /**
26       * Constructor.
27       * @param game The instance of the game
28       */
29      public GUIRowInput(Gwent game) {
30          this.game = game;
31      }
32      
33      /**
34       * Adds itself to listening for appropriate rows.
35       * @param guiInput The GUI input
36       * @param rows The rows to select from which to listen to
37       * @param card The card with which the selection is done
38       */
39      public void load(GUIInput guiInput, GUIRow[] rows, Card card) {
40          guiInput.removeListenerFromAll(this);
41          if (card instanceof UnitCard) {
42              int[] indices = ((UnitCard) card).getUnit().getTypeIndices();
43              for (int i = 0; i < indices.length; i++) {
44                  guiInput.addListener(rows[indices[i]], this);
45              }
46          } else if (card instanceof WeatherCard) {
47              Ability ability = ((WeatherCard) card).getAbilities().get(0);
48              
49              if (ability instanceof Frost) {
50                  guiInput.addListener(rows[UnitType.MELEE.getIndex()], this);
51              } else if (ability instanceof Fog) {
52                  guiInput.addListener(rows[UnitType.RANGED.getIndex()], this);
53              } else if (ability instanceof Rain) {
54                  guiInput.addListener(rows[UnitType.SIEGE.getIndex()], this);
55              } else if (ability instanceof ClearWeather) {
56                  for (int i = 0; i < UnitType.values().length; i++) {
57                      guiInput.addListener(rows[i], this);
58                  }
59              }
60          }
61      }
62      
63      @Override
64      public void releaseComponent(GUIComponent component, int button) {
65          if (component instanceof GUIRow) {
66              CombatRow row = ((GUIRow) component).getRow();
67              int index = game.getGameSystem().getBoard().getRowIndex(row);
68              game.getGameSystem().playCard(index);
69          }
70      }
71  }
