1   package fi.riissanen.gwent.game.cards.factories;

3   import fi.riissanen.gwent.game.cards.abilities.Ability;
4   import fi.riissanen.gwent.game.cards.abilities.ClearWeather;
5   import fi.riissanen.gwent.game.cards.abilities.Fog;
6   import fi.riissanen.gwent.game.cards.abilities.Frost;
7   import fi.riissanen.gwent.game.cards.abilities.Rain;
8   import fi.riissanen.gwent.game.cards.neutral.WeatherCard;
9   
10  /**
11   * A specialized {@link CardFactory} for creating {@link WeatherCard}'s.
12   * @author Daniel
13   */
14  public class WeatherCardFactory implements CardFactory {
15  
16      @Override
17      public WeatherCard createCard(CardData data) {
18          String abilityStr = data.abilities.split(",")[0];
19          Ability ability;
20          if (abilityStr.equals("Frost")) {
21              ability = new Frost();
22          } else if (abilityStr.equals("Fog")) {
23              ability = new Fog();
24          } else if (abilityStr.equals("Rain")) {
25              ability = new Rain();
26          } else if (abilityStr.equals("Clear")) {
27              ability = new ClearWeather();
28          } else {
29              return null;
30          }
31          WeatherCard card = new WeatherCard(ability);
32          return card;
33      }
34  }
