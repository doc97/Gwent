1   package fi.riissanen.gwent.game.ui;
2   
3   import fi.riissanen.gwent.engine.render.SpriteBatch;
4   import fi.riissanen.gwent.engine.render.Texture;
5   import fi.riissanen.gwent.game.cards.Card;
6   import java.util.LinkedHashMap;
7   import java.util.Map;
8   
9   /**
10   * Graphical representation of hand.
11   * @author Daniel
12   */
13  public class GUIHand extends GUIComponent {
14  
15      private final Map<Card, GUICard> cards = new LinkedHashMap<>();
16      
17      /**
18       * Creates a GUI component containing GUI cards.
19       * @param texture The texture
20       */
21      public GUIHand(Texture texture) {
22          super(texture);
23      }
24      
25      @Override
26      public void render(SpriteBatch batch) {
27          super.render(batch);
28          for (Card card : cards.keySet()) {
29              cards.get(card).render(batch);
30          }
31      }
32      
33      @Override
34      public void update() {
35          for (Card card : cards.keySet()) {
36              cards.get(card).update();
37          }
38      }
39      
40      /**
41       * Adds a GUICard, mapped to a Card key.
42       * @param card The key to map the GUI card to 
43       * @param guiCard The GUI component
44       */
45      public void addCard(Card card, GUICard guiCard) {
46          if (!cards.containsKey(card)) {
47              guiCard.setPosition(x + 10 + cards.size() * (guiCard.getWidth() + 10),
48                      y + (height - guiCard.getHeight()) / 2);
49              cards.put(card, guiCard);
50          }
51      }
52      
53      /**
54       * Remove a GUICard based on a Card key.
55       * @param card The key
56       */
57      public void removeCard(Card card) {
58          cards.get(card).destroy();
59          cards.remove(card);
60          
61          int i = 0;
62          for (Card c : cards.keySet()) {
63              GUICard guiCard = cards.get(c);
64              guiCard.setPosition(x + 10 + i * (guiCard.width + 10),
65                      y + (height - guiCard.getHeight()) / 2);
66              i++;
67          }
68      }
69      
70      /**
71       * Gets the {@link GUICard} linked to a Card.
72       * @param card The card that is linked
73       * @return The GUICard
74       */
75      public GUICard getGUICard(Card card) {
76          return cards.get(card);
77      }
78  }
