9   public class EngineLauncher {
10      
11      private Game game;
12      private final Game emptyGame = new Game() {
13          @Override
14          public void create() {
15          }
16  
17          @Override
18          public void render(double delta) {
19          }
20  
21          @Override
22          public void dispose() {
23          }
24      };
25      
26      /**
27       * Start an {@code Engine} instance with a game.
28       * @param game The game to start with the engine
29       */
30      public void start(Game game) {
31          this.game = (game == null ? emptyGame : game);
32          
33          Engine.INSTANCE.initialize();
34          this.game.create();
35          
36          double lastTime = Engine.INSTANCE.getTime();
37          int frames = 0;
38          while (!Engine.INSTANCE.display.shouldClose()) {
39              double currentTime = Engine.INSTANCE.getTime();
40              frames++;
41              double delta = currentTime - lastTime;
42              if (currentTime - lastTime >= 1.0) {
43                  Engine.INSTANCE.frameTime = 1000.0 / frames;
44                  frames = 0;
45                  lastTime += 1.0;
46              }
47              
48              this.game.render(delta);
49              Engine.INSTANCE.display.updateDisplay();
50              Engine.INSTANCE.input.update();
51          }
52          
53          this.game.dispose();
54          Engine.INSTANCE.dispose();
55      }
56  }
