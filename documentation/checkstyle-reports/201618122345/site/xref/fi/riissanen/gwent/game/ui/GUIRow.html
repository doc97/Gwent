15  /**
16   * Graphical representation of combat row.
17   *
18   * @author Daniel
19   */
20  public class GUIRow extends GUIComponent {
21  
22      private final Map<UnitCard, GUICard> cards = new HashMap<>();
23      private final CombatRow row;
24      private final TextCache cache;
25      private Text strength;
26      
27      /**
28       * Creates a GUIComponent with cards, a little like the GUI hand.
29       * @param row The row that is linked with this GUIRow
30       * @param texture The texture
31       * @param strength The text showing the rows strength
32       * @param cache The text cache where to add component texts
33       */
34      public GUIRow(CombatRow row, Texture texture, Text strength, TextCache cache) {
35          super(texture);
36          this.row = row;
37          this.cache = cache;
38          this.strength = strength;
39      }
40  
41      @Override
42      public void render(SpriteBatch batch) {
43          super.render(batch);
44          for (UnitCard card : cards.keySet()) {
45              cards.get(card).render(batch);
46          }
47      }
48      
49      @Override
50      public void update() {
51          // Update card information
52          for (Iterator<UnitCard> it = cards.keySet().iterator(); it.hasNext();) {
53              UnitCard card = it.next();
54              if (!row.hasUnit(card.getUnit())) {
55                  cards.get(card).destroy();
56                  it.remove();
57                  continue;
58              }
59              cards.get(card).update();
60          }
61          
62          // Update row strength
63          int rowStr = row.getStrength();
64          int rowStrength = Integer.parseInt(strength.getText());
65          if (rowStr != rowStrength) {
66              cache.removeText(strength);
67              Color color = strength.getColor();
68              Font font = strength.getFont();
69              float fontSize = strength.getFontSize();
70              float lineLength = strength.getMaxLineLength();
71  
72              strength = new Text(rowStr + "", font, fontSize, lineLength);;
73              strength.setColor(color.getRed(), color.getGreen(), color.getBlue());
74          }
75          
76          if (!cache.hasText(strength)) {
77              cache.addText(strength);
78              // Recalculate position after the text has been reloaded
79              strength.setPosition(x - strength.getLineLength() - 10, y + height / 2);
80          }
81      }
82      
83      @Override
84      public void setPosition(float x, float y) {
85          super.setPosition(x, y);
86          strength.setPosition(x - strength.getLineLength() - 10, y + height / 2);
87      }
88      
89      /**
90       * Adds a GUICard, mapped to a Card key.
91       * @param card The key to map the GUI card to 
92       * @param guiCard The GUI component
93       */
94      public void addCard(UnitCard card, GUICard guiCard) {
95          if (!cards.containsKey(card)) {
96              guiCard.setPosition(x + 10 + cards.size() * (guiCard.width + 10),
97                      y + (height - guiCard.getHeight()) / 2);
98              cards.put(card, guiCard);
99          }
100     }
101     
102     /**
103      * Remove a GUICard based on a Card key.
104      * @param card The key
105      */
106     public void removeCard(UnitCard card) {
107         cards.get(card).destroy();
108         cards.remove(card);
109         
110         int i = 0;
111         for (UnitCard c : cards.keySet()) {
112             GUICard guiCard = cards.get(c);
113             guiCard.setPosition(x + 10 + i * (guiCard.width + 10),
114                     y + (height - guiCard.getHeight()) / 2);
115             i++;
116         }
117     }
118     
119     public CombatRow getRow() {
120         return row;
121     }
122 }
