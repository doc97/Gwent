1   package fi.riissanen.gwent.game.cards.abilities;
2   
3   import fi.riissanen.gwent.game.GameSystem;
4   import fi.riissanen.gwent.game.cards.UnitCard;
5   import fi.riissanen.gwent.game.combat.Unit;
6   import fi.riissanen.gwent.game.states.GameStates;
7   
8   /**
9    * Medic ability
10   * @author Daniel
11   */
12  public class Medic implements Ability, UnitAbility {
13  
14      private final UnitCard card;
15      
16      public Medic(UnitCard card) {
17          this.card = card;
18      }
19      
20      @Override
21      public void activate(GameSystem system) {
22          system.getStateSystem().push(GameStates.DISCARD_PILE_STATE);
23      }
24      
25      
26      @Override
27      public Unit getUnit() {
28          return card.getUnit();
29      }
30  
31      @Override
32      public String getName() {
33          return "Medic";
34      }
35  
36      @Override
37      public String getDescription() {
38          return "Choose a card from your discard pile and play it instantly. It"
39                  + "cannot be a hero card or a special card";
40      }
41  }
