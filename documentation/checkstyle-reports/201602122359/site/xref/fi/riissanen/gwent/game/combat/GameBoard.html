1   package fi.riissanen.gwent.game.combat;
2   
3   import fi.riissanen.gwent.engine.Engine;
4   import fi.riissanen.gwent.engine.Logger.LogLevel;
5   import fi.riissanen.gwent.game.cards.neutral.WeatherCard;
6   import java.util.ArrayList;
7   import java.util.Iterator;
8   import java.util.List;
9   
10  /**
11   * A class representing the game board
12   *
13   * @author Daniel
14   */
15  public class GameBoard {
16  
17      private final CombatRow[] friendlyRows;
18      private final CombatRow[] enemyRows;
19      private final List<WeatherCard> weatherPile;
20      private final List<Unit> savedUnits;
21  
22      public GameBoard() {
23          friendlyRows = new CombatRow[]{
24              new CombatRow(), new CombatRow(), new CombatRow()
25          };
26          enemyRows = new CombatRow[]{
27              new CombatRow(), new CombatRow(), new CombatRow()
28          };
29          weatherPile = new ArrayList<>();
30          savedUnits = new ArrayList<>();
31      }
32  
33      public void addUnit(Unit unit, UnitType type, boolean isFriendly) {
34          if (unit == null) {
35              return;
36          }
37          
38          if (!unit.getTypes().contains(type)) {
39              throw new IllegalArgumentException("Unit is not of unit type " + type);
40          }
41          
42          CombatRow[] rows = getRows(isFriendly);
43          int index = type.getIndex();
44          rows[index].addUnit(unit);
45      }
46      
47      
48      
49      public void addWeatherCard(WeatherCard weather) {
50          if (weather != null) {
51              weatherPile.add(weather);
52          }
53      }
54      
55      public void clearWeather() {
56          for (WeatherCard card : weatherPile) {
57              card.getOwner().discardCard(card);
58          }
59          weatherPile.clear();
60      }
61      
62      public void clearRows() {
63          for (CombatRow row : getRows(true)) {
64              for (Iterator<Unit> it = row.getUnits().iterator(); it.hasNext();) {
65                  Unit unit = it.next();
66                  if (!savedUnits.contains(unit)) {
67                      it.remove();
68                  }
69              }
70          }
71          savedUnits.clear();
72      }
73      
74      public void saveUnit(Unit u) {
75          savedUnits.add(u);
76      }
77      
78      public void status() {
79          Engine.INSTANCE.log.write(LogLevel.INFO, "Board state");
80          
81          String rowName = "";
82          int strength = 0;
83          int rowCount = UnitType.values().length;
84          for (int i = 0; i < rowCount * 2; i++) {
85              boolean friendly = (i >= rowCount);
86              UnitType row = UnitType.values()[i % rowCount];
87              rowName = row.toString();
88              strength = getRowStrength(row, friendly);
89              Engine.INSTANCE.log.write(LogLevel.INFO, rowName + ": " + strength);
90              
91              if (i == rowCount - 1) {
92                  Engine.INSTANCE.log.write(LogLevel.INFO, "---");
93              }
94          }
95      }
96  
97      public int getRowStrength(UnitType row, boolean friendly) {
98          return getRow(friendly, row).getStrength();
99      }
100     
101     public int getStrength(boolean friendly) {
102         int sum = 0;
103         for (CombatRow row : getRows(friendly)) {
104             sum += row.getStrength();
105         }
106         return sum;
107     }
108     
109     public int getUnitCount(boolean friendly) {
110         int sum = 0;
111         for (CombatRow row : getRows(friendly)) {
112             sum += row.getUnitCount();
113         }
114         return sum;
115     }
116     
117     public int getTotalUnitCount() {
118         int friendlyCount = getUnitCount(true);
119         int enemyCount = getUnitCount(false);
120         return friendlyCount + enemyCount;
121     }
122 
123     public int getWeatherCardCount() {
124         return weatherPile.size();
125     }
126     
127     public int getSavedUnitCount() {
128         return savedUnits.size();
129     }
130     
131     public CombatRow[] getRows(boolean friendly) {
132         return friendly ? friendlyRows : enemyRows;
133     }
134     
135     public CombatRow getRow(boolean friendly, UnitType row) {
136         return friendly ? friendlyRows[row.getIndex()] : enemyRows[row.getIndex()];
137     }
138     
139     public CombatRow getRow(Unit unit) {
140         for (CombatRow row : getRows(unit.isFriendly())) {
141             if (row.hasUnit(unit)) {
142                 return row;
143             }
144         }
145         return null;
146     }
147     
148     public int getRowIndex(CombatRow row) {
149         CombatRow[] fRows = getRows(true);
150         CombatRow[] eRows = getRows(false);
151         for (int i = 0; i < fRows.length; i++) {
152             if (fRows[i].equals(row) || eRows[i].equals(row)) {
153                 return i;
154             }
155         }
156         return -1;
157     }
158 }
