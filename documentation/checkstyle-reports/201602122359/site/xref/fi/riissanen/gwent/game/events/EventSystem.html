1   package fi.riissanen.gwent.game.events;
2   
3   import java.util.ArrayDeque;
4   import java.util.HashMap;
5   import java.util.List;
6   import java.util.Map;
7   import java.util.Queue;
8   
9   /**
10   * Handles event distribution and listeners
11   * @author Daniel
12   */
13  public class EventSystem {
14  
15      private final Map<Class<? extends Event>, List<EventListener>> listeners = new HashMap<>();
16      private final Queue<Event> events = new ArrayDeque<>();
17      
18      public void register(Event event) {
19          events.add(event);
20      }
21      
22      public void update() {
23          Event e;
24          while ((e = events.poll()) != null) {
25              handleEvent(e);
26          }
27      }
28      
29      private void handleEvent(Event event) {
30          List<EventListener> eventListeners = listeners.get(event.getClass());
31          if (eventListeners != null) {
32              for (EventListener listener : eventListeners) {
33                  listener.process(event);
34              }
35          }
36      }
37      
38      public void clear() {
39          events.clear();
40      }
41      
42      public void setListeners(Class<? extends Event> clazz, List<EventListener> listeners) {
43          this.listeners.put(clazz, listeners);
44      }
45      
46      public int getEventsPendingCount() {
47          return events.size();
48      }
49      
50      public int getEventListenerCount(Class<? extends Event> clazz) {
51          return listeners.get(clazz).size();
52      }
53  }
