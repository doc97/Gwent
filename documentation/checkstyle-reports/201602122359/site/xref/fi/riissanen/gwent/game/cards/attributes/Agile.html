14      @Override
15      public void activate(Unit unit) {
16          List<UnitType> types = new ArrayList<>();
17          types.add(UnitType.MELEE);
18          types.add(UnitType.RANGED);
19          unit.setUnitTypes(types);
20      }
21      
22      @Override
23      public String getName() {
24          return "Agile";
25      }
26

28      public String getDescription() {

30      }
