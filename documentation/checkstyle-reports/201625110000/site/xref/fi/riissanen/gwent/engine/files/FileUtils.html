8   import java.util.List;

11   *
12   * @author Daniel
13   */
14  public class FileUtils {
15      
16      public List<String> readLines(String filename)
17              throws FileNotFoundException, IOException {
18          List<String> lines = new ArrayList<>();
19          BufferedReader br = new BufferedReader(new FileReader(filename));
20          String line;
21          while ((line = br.readLine()) != null) {
22              lines.add(line);
23          }
24          return lines;
25      }
26  }
