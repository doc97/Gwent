1   package fi.riissanen.gwent.game.cards;
2   
3   import java.util.ArrayList;
4   import java.util.Collection;
5   import java.util.List;
6   
7   /**
8    * An abstract data structure for {@link Card}s
9    * @author Daniel
10   */
11  public abstract class CardCollection {
12      
13      private final List<Card> cards = new ArrayList<>();
14  
15      public void addCard(Card card) {
16          if (card != null) {
17              cards.add(card);
18          }
19      }
20  
21      public void addCards(Collection<? extends Card> cards) {
22          for (Card card : cards) {
23              addCard(card);
24          }
25      }
26      
27      public boolean removeCard(Card card) {
28          return cards.remove(card);
29      }
30      
31      public boolean containsCard(Card card) {
32          return cards.contains(card);
33      }
34  
35      public Card getCard(int index) {
36          if (index < 0 || index >= cards.size()) {
37              return null;
38          }
39          return cards.get(index);
40      }
41  
42      public int getCardCount() {
43          return cards.size();
44      }
45  }
