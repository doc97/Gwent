1   package fi.riissanen.gwent.game.cards.abilities;
2   
3   import fi.riissanen.gwent.game.GameSystem;
4   import fi.riissanen.gwent.game.cards.Card;
5   import fi.riissanen.gwent.game.cards.UnitCard;
6   import fi.riissanen.gwent.game.combat.Unit;
7   import java.util.ArrayList;
8   import java.util.List;
9   
10  /**
11   * Tight bond ability
12   * @author Daniel
13   */
14  
15  public class TightBond implements Ability, UnitAbility {
16  
17      private final Card card;
18      
19      public TightBond(Card card) {
20          this.card = card;
21      }
22      
23      @Override
24      public void activate(GameSystem system) {
25          List<Unit> units = system.getBoard().getRow(getUnit()).getUnits();
26          List<Unit> affectedUnits = new ArrayList<>();
27          for (Unit rowUnit : units) {
28              if (rowUnit.getName().equals(getUnit().getName())) {
29                  affectedUnits.add(rowUnit);
30              }
31          }
32  
33          for (Unit unit : affectedUnits) {
34              int totalStr = unit.getBaseStrength() * (int) Math.pow(2, affectedUnits.size() - 1);
35              unit.setEffectStrength(totalStr - unit.getBaseStrength());
36          }
37      }
38      
39      @Override
40      public Unit getUnit() {
41          if (card instanceof UnitCard) {
42              return ((UnitCard) card).getUnit();
43          }
44          return null;
45      }
46  
47      @Override
48      public String getName() {
49          return "Tight bond";
50      }
51  
52      @Override
53      public String getDescription() {
54          return "Double the strength of units in the same row with the same name";
55      }
56  
57  }
