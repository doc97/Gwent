1   package fi.riissanen.gwent.game.factions;
2   
3   import fi.riissanen.gwent.game.GameSystem;
4   import fi.riissanen.gwent.game.Player;
5   import fi.riissanen.gwent.game.cards.abilities.Ability;
6   import fi.riissanen.gwent.game.combat.CombatRow;
7   import fi.riissanen.gwent.game.combat.Unit;
8   import fi.riissanen.gwent.game.events.Event;
9   import fi.riissanen.gwent.game.events.RoundEndEvent;
10  import java.util.Random;
11  
12  /**
13   * A faction in the game
14   * @author Daniel
15   */
16  public class Monsters extends Faction {
17  
18      private final Ability ability;
19      private boolean triggered;
20      
21      public Monsters(Player player) {
22          ability = new Ability() {
23              @Override
24              public void activate(GameSystem system) {
25                  Random rand = new Random();
26                  int rowIndex = rand.nextInt(3);
27                  CombatRow row = system.getBoard().getRows(player.isFriendly())[rowIndex];
28                  int unitIndex = rand.nextInt(row.getUnitCount());
29                  Unit unit = row.getUnits().get(unitIndex);
30                  system.getBoard().saveUnit(unit);
31              }
32  
33              @Override
34              public String getName() {
35                  return "Monsters ability";
36              }
37  
38              @Override
39              public String getDescription() {
40                  return "Keeps one random unit card on the battlefield after each round";
41              }
42          };
43      }
44      
45      @Override
46      public Ability getAbility() {
47          return ability;
48      }
49  
50      @Override
51      public boolean isTriggered() {
52          return triggered;
53      }
54  
55      @Override
56      public void process(Event event) {
57          if (event instanceof RoundEndEvent) {
58              triggered = true;
59          }
60      }
61  
62  }
