66      /**
67       * Initializes with a default shader.
68       * @param defaultShader The default shader
69       */
70      public SpriteBatch(ShaderProgram defaultShader) {
71          indices = new int[6000];
72          vertices = new float[12000];
73          texCoords = new float[8000];
74          colors = new float[16000];
75          vbos = new ArrayList<>();
76          intBuff = BufferUtils.createIntBuffer(6000);
77          floatBuff1 = BufferUtils.createFloatBuffer(12000);
78          floatBuff2 = BufferUtils.createFloatBuffer(8000);
79          floatBuff3 = BufferUtils.createFloatBuffer(16000);
80          color = new Color(1, 1, 1, 1);
81          viewport = new Viewport();
82  
83          for (int i = 0, j = 0; i < 6000; i += 6, j += 4) {
84              indices[i    ] = j;
85              indices[i + 1] = (j + 1);
86              indices[i + 2] = (j + 2);
87              indices[i + 3] = (j + 2);
88              indices[i + 4] = (j + 3);
89              indices[i + 5] = j;
90          }
91          vao = glGenVertexArrays();
92          ibo = glGenBuffers();
93          
94          if (defaultShader == null) {
95              this.defaultShader = createDefaultShader();
96              createdShader = true;
97          } else {
98              this.defaultShader = defaultShader;
99          }
100     }

102     private ShaderProgram createDefaultShader() {
103         ShaderProgram shader = new DefaultShader();
104         if (!shader.isCompiled()) {
105             throw new IllegalArgumentException(
106                     "Error compiling shader: " + shader.getLog());
107         }
108         return shader;
109     }

111     /**
112      * Starts a new batch of draw calls.
113      * 
114      * <p>
115      * Must not be called after another begin() call without an end() call in
116      * between. Must be called before a draw() call.
117      */
118     public void begin() {
119         if (isDrawing) {
120             throw new IllegalStateException(
121                     "SpriteBatch#end() must be called before begin.");
122         }
123         isDrawing = true;
124         if (customShader == null) {
125             defaultShader.start();
126         } else {
127             customShader.start();
128         }
129     }
130 
131     /**
132      * Ends the batching and flushes the batch.
133      */
134     public void end() {
135         if (!isDrawing) {
136             throw new IllegalStateException(
137                 "SpriteBatch#begin() must be called before end.");
138         }
139         flush();
140         
141         if (customShader == null) {
142             defaultShader.start();
143         } else {
144             customShader.start();
145         }
146         
147         lastTexture = null;
148         isDrawing = false;
149     }
150 
151     /**
152      * Flushes the batch of draw calls to the graphics card.
153      */
154     public void flush() {
155         if (idx == 0) {
156             return;
157         }
158         
159         int sprites = idx / 12;
160         
161         bindVAO();
162         loadToVAO(vertices, texCoords, colors, indices);
163         glEnableVertexAttribArray(0);
164         glEnableVertexAttribArray(1);
165         glEnableVertexAttribArray(2);
166         glEnableVertexAttribArray(3);
167         glDrawElements(GL_TRIANGLES, sprites * 6, GL_UNSIGNED_INT, 0);
168         glDisableVertexAttribArray(0);
169         glDisableVertexAttribArray(1);
170         glDisableVertexAttribArray(2);
171         glDisableVertexAttribArray(3);
172 
173         unbindVAO();
174         glDeleteBuffers(ibo);
175         for (int vbo : vbos) {
176             glDeleteBuffers(vbo);
177         }
178         vbos.clear();
179         idx = 0;
180     }

182     private void changeTexture(Texture texture) {
183         flush();
184         if (texture != null) {
185             lastTexture = texture;
186             glBindTexture(GL_TEXTURE_2D, texture.getID());
187         } else {
188             lastTexture = null;
189             glBindTexture(GL_TEXTURE_2D, 0);
190         }
191     }

215     /**
216      * Draws a texture with a size, rotation and specified UV coordinates.
217      * @param texture The texture
218      * @param x The x position (upper-left)
219      * @param y The y position (upper-left)
220      * @param width The width to draw the texture as
221      * @param height The height to draw the texture as
222      * @param uvs UV coordinates, can be used to draw a part of the texture
223      * @param rotation Rotation in radians
224      * @param anchorPX The point of origin around which to rotate
225      * @param anchorPY The point of origin around which to rotate
226      */
227     public void draw(Texture texture, float x, float y, float width, float height,
228                     float[] uvs, float rotation, float anchorPX, float anchorPY) {
229         Color[] currColorArr = { color, color, color, color };
230         draw(texture, x, y + height, x + width, y + height, x + width, y, x, y, uvs,
231                 currColorArr, rotation, anchorPX, anchorPY);
232     }

234     /**
235      * Draws a texture with a 4 vertex polygon, rotation, UV coordinates and colors.
236      * @param texture The texture
237      * @param x1 Vertex 1
238      * @param y1 Vertex 1
239      * @param x2 Vertex 2
240      * @param y2 Vertex 2
241      * @param x3 Vertex 3
242      * @param y3 Vertex 3
243      * @param x4 Vertex 4
244      * @param y4 Vertex 4
245      * @param uvs UV coordinates, can be used to draw a part of the texture
246      * @param colours The colors for each vertex (should contain four elements)
247      * @param rotation Rotation in radians
248      * @param anchorPX The point of origin around which to rotate
249      * @param anchorPY The point of origin around which to rotate
250      */
251     public void draw(Texture texture, float x1, float y1,
252             float x2, float y2, float x3, float y3, float x4, float y4,
253             float[] uvs, Color[] colours, float rotation,
254             float anchorPX, float anchorPY) {
255 
256         if (!isDrawing) {
257             throw new IllegalStateException(
258                 "SpriteBatch#begin() must be called before draw.");
259         }
260         
261         if (texture != lastTexture) {
262             changeTexture(texture);
263         }
264         
265         if (idx == vertices.length) {
266             flush();
267         }
268 
269         // To viewport coordinates
270         x1 = viewport.toGlCoordinateX(x1);
271         y1 = viewport.toGlCoordinateY(y1);
272         x2 = viewport.toGlCoordinateX(x2);
273         y2 = viewport.toGlCoordinateY(y2);
274         x3 = viewport.toGlCoordinateX(x3);
275         y3 = viewport.toGlCoordinateY(y3);
276         x4 = viewport.toGlCoordinateX(x4);
277         y4 = viewport.toGlCoordinateY(y4);
278         
279         //coords for the vertices
280         float vx1 = 2 * scale * x1 - (1 - scale);
281         float vy1 = 2 * scale * y1 - (1 - scale);
282         float vx2 = 2 * scale * x2 - (1 - scale);
283         float vy2 = 2 * scale * y2 - (1 - scale);
284         float vx3 = 2 * scale * x3 - (1 - scale);
285         float vy3 = 2 * scale * y3 - (1 - scale);
286         float vx4 = 2 * scale * x4 - (1 - scale);
287         float vy4 = 2 * scale * y4 - (1 - scale);
288 
289         int tempidx = this.idx;
290         int vertexCount = tempidx / 3;
291         int cdx = vertexCount << 2;
292         int tdx = vertexCount << 1;
293 
294         if (rotation != 0) {
295             float p1d = (float) Math.hypot(anchorPX - vx1, anchorPY - vy1);
296             float p2d = (float) Math.hypot(anchorPX - vx2, anchorPY - vy2);
297             float p3d = (float) Math.hypot(anchorPX - vx3, anchorPY - vy3);
298             float p4d = (float) Math.hypot(anchorPX - vx4, anchorPY - vy4);
299 
300             double angle = Math.atan2(vy1 - anchorPY, vx1 - anchorPX);
301             vx1 = (float) (anchorPX + Math.cos(rotation + angle) * p1d);
302             vy1 = (float) (anchorPY + Math.sin(rotation + angle) * p1d);
303 
304             angle = Math.atan2(vy2 - anchorPY, vx2 - anchorPX);
305             vx2 = (float) (anchorPX + Math.cos(rotation + angle) * p2d);
306             vy2 = (float) (anchorPY + Math.sin(rotation + angle) * p2d);
307 
308             angle = Math.atan2(vy3 - anchorPY, vx3 - anchorPX);
309             vx3 = (float) (anchorPX + Math.cos(rotation + angle) * p3d);
310             vy3 = (float) (anchorPY + Math.sin(rotation + angle) * p3d);
311 
312             angle = Math.atan2(vy4 - anchorPY, vx4 - anchorPX);
313             vx4 = (float) (anchorPX + Math.cos(rotation + angle) * p4d);
314             vy4 = (float) (anchorPY + Math.sin(rotation + angle) * p4d);
315         }
316 
317         vertices[tempidx++] = vx1;
318         vertices[tempidx++] = vy1;
319         vertices[tempidx++] = 0;
320         vertices[tempidx++] = vx2;
321         vertices[tempidx++] = vy2;
322         vertices[tempidx++] = 0;
323         vertices[tempidx++] = vx3;
324         vertices[tempidx++] = vy3;
325         vertices[tempidx++] = 0;
326         vertices[tempidx++] = vx4;
327         vertices[tempidx++] = vy4;
328         vertices[tempidx++] = 0;
329 
330         if (texture != null) {
331             texCoords[tdx++] = uvs[0];
332             texCoords[tdx++] = uvs[1];
333 
334             texCoords[tdx++] = uvs[2];
335             texCoords[tdx++] = uvs[1];
336 
337             texCoords[tdx++] = uvs[2];
338             texCoords[tdx++] = uvs[3];
339 
340             texCoords[tdx++] = uvs[0];
341             texCoords[tdx++] = uvs[3];
342         }
343 
344         for (int i = 0; i < 4; i++) {
345             colors[cdx++] = colours[i].getRed();
346             colors[cdx++] = colours[i].getGreen();
347             colors[cdx++] = colours[i].getBlue();
348             colors[cdx++] = colours[i].getAlpha();
349         }
350 
351         this.idx = tempidx;
352     }

354     private void loadToVAO(float[] positions, float[] texCoords, float[] colors, int[] indices) {
355         bindIndicesBuffer(indices, intBuff);
356         storeDataInAttributeList(0, floatBuff1, 3, positions);
357         storeDataInAttributeList(1, floatBuff2, 2, texCoords);
358         storeDataInAttributeList(2, floatBuff3, 4, colors);
359     }
360 
361     private void bindVAO() {
362         glBindVertexArray(vao);
363     }
364 
365     private void unbindVAO() {
366         glBindVertexArray(0);
367     }
368 
369     private void bindIndicesBuffer(int[] indices, IntBuffer buffer) {
370         ibo = glGenBuffers();
371         glBindBuffer(GL_ELEMENT_ARRAY_BUFFER, ibo);
372         storeDataInIntBuffer(buffer, indices);
373         glBufferData(GL_ELEMENT_ARRAY_BUFFER, buffer, GL_STATIC_DRAW);
374     }
375 
376     private IntBuffer storeDataInIntBuffer(IntBuffer buffer, int[] data) {
377         buffer.clear();
378         buffer.put(data);
379         buffer.flip();
380         return buffer;
381     }
382 
383     private void storeDataInAttributeList(int attribute, FloatBuffer buffer, int size, float[] data) {
384         int vbo = glGenBuffers();
385         vbos.add(vbo);
386         glBindBuffer(GL_ARRAY_BUFFER, vbo);
387         storeDataInFloatBuffer(buffer, data);
388         glBufferData(GL_ARRAY_BUFFER, buffer, GL_STATIC_DRAW);
389         glVertexAttribPointer(attribute, size, GL_FLOAT, false, 0, 0);
390         glBindBuffer(GL_ARRAY_BUFFER, 0);
391     }
392 
393     private FloatBuffer storeDataInFloatBuffer(FloatBuffer buffer, float[] data) {
394         buffer.clear();
395         buffer.put(data);
396         buffer.flip();
397         return buffer;
398     }

400     /**
401      * Releases all allocated resources.
402      */
403     public void dispose() {
404         glDeleteBuffers(ibo);
405         for (int vbo : vbos) {
406             glDeleteBuffers(vbo);
407         }
408         vbos.clear();
409 
410         BufferUtils.zeroBuffer(intBuff);
411         BufferUtils.zeroBuffer(floatBuff1);
412         BufferUtils.zeroBuffer(floatBuff2);
413         BufferUtils.zeroBuffer(floatBuff3);
414         
415         if (createdShader && customShader != null) {
416             defaultShader.dispose();
417         }
418     }

458     /**
459      * Sets a new custom shader.
460      * @param shader The new shader
461      */
462     public void setShader(ShaderProgram shader) {
463         if (isDrawing) {
464             flush();
465             if (customShader == null) {
466                 defaultShader.stop();
467             } else {
468                 customShader.stop();
469             }
470         }
471         
472         customShader = shader;
473         
474         if (isDrawing) {
475             flush();
476             if (customShader == null) {
477                 defaultShader.start();
478             } else {
479                 customShader.start();
480             }
481         }
482     }
