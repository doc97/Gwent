1   package fi.riissanen.gwent.game.combat;
2   
3   import fi.riissanen.gwent.game.cards.abilities.Ability;
4   import java.util.ArrayList;
5   import java.util.List;
6   
7   /**
8    * A class representing a combat row on the game board.
9    *
10   * @author Daniel
11   * @see GameBoard
12   */
13  public class CombatRow {
14  
15      private final List<Unit> units = new ArrayList<>();
16  
17      /**
18       * Adds a unit to the row as long as it is not null.
19       * @param unit The unit to add
20       */
21      public void addUnit(Unit unit) {
22          if (unit != null) {
23              units.add(unit);
24          }
25      }
26  
27      /**
28       * Gets the combined strength of all units in the row.
29       * @return The combined strength
30       */
31      public int getStrength() {
32          int sum = 0;
33          for (Unit unit : units) {
34              sum += unit.getStrength();
35          }
36          return sum;
37      }
38      
39      /**
40       * Lists all units in the row with a certain ability.
41       * @param clazz The class of the ability
42       * @return The list of units
43       */
44      public List<Unit> getUnitsWithAbility(Class<? extends Ability> clazz) {
45          List<Unit> abilityUnits = new ArrayList<>();
46          for (Unit u : units) {
47              if (u.hasAbility(clazz)) {
48                  abilityUnits.add(u);
49              }
50          }
51          return abilityUnits;
52      }
53  
54      public List<Unit> getUnits() {
55          return units;
56      }
57      
58      /**
59       * Returns the unit count in the row.
60       * @return The unit count
61       */
62      public int getUnitCount() {
63          return units.size();
64      }
65      
66      /**
67       * Checks whether the row contains a unit.
68       * @param unit The unit to check
69       * @return True if the row contains the unit
70       */
71      public boolean hasUnit(Unit unit) {
72          return units.contains(unit);
73      }
74  }
