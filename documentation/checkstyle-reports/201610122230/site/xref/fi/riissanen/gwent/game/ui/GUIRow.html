1   package fi.riissanen.gwent.game.ui;
2   
3   import fi.riissanen.gwent.engine.render.SpriteBatch;
4   import fi.riissanen.gwent.engine.render.Texture;
5   import fi.riissanen.gwent.game.cards.Card;
6   import java.util.HashMap;
7   import java.util.Map;
8   
9   /**
10   * Graphical representation of combat row.
11   *
12   * @author Daniel
13   */
14  public class GUIRow extends GUIComponent {
15  
16      private final Map<Card, GUICard> cards = new HashMap<>();
17      private final TextCache cache;
18  
19      /**
20       * Creates a GUIComponent with cards, a little like the GUI hand.
21       * @param texture The texture
22       * @param cache The text cache where to add component texts
23       */
24      public GUIRow(Texture texture, TextCache cache) {
25          super(texture);
26          this.cache = cache;
27      }
28  
29      @Override
30      public void render(SpriteBatch batch) {
31          super.render(batch);
32          for (Card card : cards.keySet()) {
33              cards.get(card).render(batch);
34          }
35      }
36      
37      /**
38       * Adds a GUICard, mapped to a Card key.
39       * @param card The key to map the GUI card to 
40       * @param guiCard The GUI component
41       */
42      public void addCard(Card card, GUICard guiCard) {
43          if (!cards.containsKey(card)) {
44              guiCard.setPosition(x + 10 + cards.size() * (guiCard.width + 10),
45                      y + (height - guiCard.getHeight()) / 2);
46              cards.put(card, guiCard);
47              cache.addText(guiCard.getText());
48          }
49      }
50      
51      /**
52       * Remove a GUICard based on a Card key.
53       * @param card The key
54       */
55      public void removeCard(Card card) {
56          cache.removeText(cards.get(card).getText());
57          cards.remove(card);
58          
59          int i = 0;
60          for (Card c : cards.keySet()) {
61              GUICard guiCard = cards.get(c);
62              guiCard.setPosition(x + 10 + i * (guiCard.width + 10),
63                      y + (height - guiCard.getHeight()) / 2);
64              i++;
65          }
66      }
67  }
