1   package fi.riissanen.gwent.engine.render.fonts;
2   
3   import java.util.ArrayList;
4   import java.util.List;
5   
6   /**
7    * Contains data about a {@code Text}'s mesh.
8    * @author Daniel
9    */
10  public class TextMeshData {
11      
12      private final List<TextLine> lines = new ArrayList<>();
13      private final List<Float> quadData = new ArrayList<>();
14      private final List<Float> uvs = new ArrayList<>();
15      
16      private float[] quadDataArr;
17      private float[] uvArr;
18      
19      /**
20       * Creates the mesh data.
21       * @param text The text to create the data of
22       */
23      public void create(Text text) {
24          clear();
25          calculateLineStructure(text);
26          calculateMesh(text, lines);
27          storeToArrays();
28      }
29      
30      private void clear() {
31          lines.clear();
32          quadData.clear();
33          uvs.clear();
34          quadDataArr = new float[0];
35          uvArr = new float[0];
36      }
37      private void calculateLineStructure(Text text) {
38          char[] chars = text.getText().toCharArray();
39          Font font = text.getFont();
40          TextLine currLine = new TextLine(
41                  font.getSpaceWidth(),
42                  text.getFontSize(),
43                  text.getLineLength());
44          TextWord currWord = new TextWord(text.getFontSize());
45          for (char c : chars) {
46              int ascii = (int) c;
47              if (ascii == Font.ASCII_SPACE) {
48                  addWordToLine(lines, currLine, currWord, text);
49                  currWord = new TextWord(text.getFontSize());
50                  continue;
51              }
52              Glyph character = text.getFont().getGlyph(ascii);
53              currWord.addCharacter(character);
54          }
55          addWordToLine(lines, currLine, currWord, text);
56          lines.add(currLine);
57      }
58      
59      private void addWordToLine(List<TextLine> lines, TextLine currLine, TextWord currWord, Text text) {
60          boolean added = currLine.addWord(currWord);
61          // Create new line
62          if (!added) {
63              lines.add(currLine);
64              currLine = new TextLine(
65                      text.getFont().getSpaceWidth(),
66                      text.getFontSize(),
67                      text.getLineLength());
68              currLine.addWord(currWord);
69          }
70      }
71      
72      private void calculateMesh(Text text, List<TextLine> lines) {
73          text.setNumberOfLines(lines.size());
74          double cursorX = 0;
75          double cursorY = 0;
76          for (TextLine line : lines) {
77              for (TextWord word : line.getWords()) {
78                  for (Glyph glyph : word.getCharacters()) {
79                      addGlyphVertices(cursorX, cursorY, glyph,
80                              text.getFontSize(), text.getFont().getLineHeight());
81                      addGlyphUVs(glyph.getU1(), glyph.getV1(), glyph.getU2(), glyph.getV2());
82                      cursorX += glyph.getXAdvance() * text.getFontSize();
83                  }
84                  cursorX += text.getFont().getSpaceWidth() * text.getFontSize();
85              }
86              cursorX = 0;
87              cursorY -= text.getFont().getLineHeight() * text.getFontSize();
88          }
89      }
90      
91      private void addGlyphVertices(double cursorX, double cursorY, Glyph glyph,
92              double fontSize, double lineHeight) {
93          double width = glyph.getWidth() * fontSize;
94          double height = glyph.getHeight() * fontSize;
95          double x = cursorX + glyph.getXOffset() * fontSize;
96          double y = lineHeight * fontSize - cursorY - glyph.getYOffset() * fontSize - height;
97          addQuadData(x, y, width, height);
98      }
99      
100     private void addQuadData(double x, double y, double width, double height) {
101         quadData.add((float) x);
102         quadData.add((float) y);
103         quadData.add((float) width);
104         quadData.add((float) height);
105     }
106     
107     private void addGlyphUVs(double u1, double v1, double u2, double v2) {
108         uvs.add((float) u1);
109         uvs.add((float) v1);
110         uvs.add((float) u2);
111         uvs.add((float) v2);
112     }
113     
114     private void storeToArrays() {
115         quadDataArr = floatListToArray(quadData);
116         uvArr = floatListToArray(uvs);
117     }
118     
119     private float[] floatListToArray(List<Float> list) {
120         float[] arr = new float[list.size()];
121         for (int i = 0; i < arr.length; i++) {
122             arr[i] = list.get(i);
123         }
124         return arr;
125     }
126     
127     public float[] getQuadData() {
128         return quadDataArr;
129     }
130     
131     public float[] getUVs() {
132         return uvArr;
133     }
134     
135     public int getVertexCount() {
136         return quadDataArr.length;
137     }
138     
139     public int getGlyphCount() {
140         return getVertexCount() / 4;
141     }
142 }
