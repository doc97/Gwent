1   package fi.riissanen.gwent.game.cards;
2   
3   import java.util.ArrayList;
4   import java.util.Collection;
5   import java.util.List;
6   
7   /**
8    * An abstract data structure for {@link Card}s.
9    * @author Daniel
10   */
11  public abstract class CardCollection {
12      
13      private final List<Card> cards = new ArrayList<>();
14  
15      /**
16       * Adds a card to the collection, unless null.
17       * @param card The card to add
18       */
19      public void addCard(Card card) {
20          if (card != null) {
21              cards.add(card);
22          }
23      }
24  
25      /**
26       * Adds a collection of cards.
27       * @param cards The collection
28       */
29      public void addCards(Collection<? extends Card> cards) {
30          for (Card card : cards) {
31              addCard(card);
32          }
33      }
34      
35      /**
36       * Clears the card collection.
37       */
38      public void clearCards() {
39          cards.clear();
40      }
41      
42      /**
43       * Removes a card from the collection.
44       * @param card The card to remove
45       * @return The success of the removal
46       */
47      public boolean removeCard(Card card) {
48          return cards.remove(card);
49      }
50      
51      /**
52       * Checks whether a collection contains the card.
53       * @param card The card to check
54       * @return True if the collection contains the card
55       */
56      public boolean containsCard(Card card) {
57          return cards.contains(card);
58      }
59  
60      /**
61       * Gets a card based on it's index.
62       * @param index The card index
63       * @return The card or null if the index is out of bounds
64       */
65      public Card getCard(int index) {
66          if (index < 0 || index >= cards.size()) {
67              return null;
68          }
69          return cards.get(index);
70      }
71  
72      public int getCardCount() {
73          return cards.size();
74      }
75  }
