9       private static final String VERTEX_SHADER = "#version 400\n" //
10          + "layout(location = 0) in vec3 " + ShaderProgram.POSITION_ATTRIBUTE + ";\n" //
11          + "layout(location = 1) in vec2 " + ShaderProgram.TEXCOORD_ATTRIBUTE + ";\n" //

14          + "out vec2 v_texCoords;\n" //
15          + "\n" //

17          + "{\n" //
18          + "   v_color = " + ShaderProgram.COLOR_ATTRIBUTE + ";\n" //

20          + "   v_texCoords = " + ShaderProgram.TEXCOORD_ATTRIBUTE + ";\n" //
21          + "   gl_Position = vec4(" + ShaderProgram.POSITION_ATTRIBUTE + ", 1.0);\n" //
22          + "}\n";

24          + "in vec4 v_color;\n" //
25          + "in vec2 v_texCoords;\n" //
26          + "out vec4 frag_color;\n" //
27          + "uniform sampler2D u_texture;\n" //
28          + "void main()\n"//
29          + "{\n" //
30          + "  frag_color = v_color * texture(u_texture, v_texCoords);\n" //
31          + "}";
32      
33      /**
