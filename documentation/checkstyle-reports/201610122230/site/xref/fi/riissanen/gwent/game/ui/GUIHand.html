1   package fi.riissanen.gwent.game.ui;
2   
3   import fi.riissanen.gwent.engine.render.SpriteBatch;
4   import fi.riissanen.gwent.engine.render.Texture;
5   import fi.riissanen.gwent.game.cards.Card;
6   import java.util.HashMap;
7   import java.util.Map;
8   
9   /**
10   * Graphical representation of hand.
11   * @author Daniel
12   */
13  public class GUIHand extends GUIComponent {
14  
15      private final Map<Card, GUICard> cards = new HashMap<>();
16      private final TextCache cache;
17      
18      /**
19       * Creates a GUI component containing GUI cards.
20       * @param texture The texture
21       * @param cache The text cache to add component texts to
22       */
23      public GUIHand(Texture texture, TextCache cache) {
24          super(texture);
25          this.cache = cache;
26      }
27      
28      @Override
29      public void render(SpriteBatch batch) {
30          super.render(batch);
31          for (Card card : cards.keySet()) {
32              cards.get(card).render(batch);
33          }
34      }
35      
36      /**
37       * Adds a GUICard, mapped to a Card key.
38       * @param card The key to map the GUI card to 
39       * @param guiCard The GUI component
40       */
41      public void addCard(Card card, GUICard guiCard) {
42          if (!cards.containsKey(card)) {
43              guiCard.setPosition(x + 10 + cards.size() * (guiCard.getWidth() + 10),
44                      y + (height - guiCard.getHeight()) / 2);
45              cards.put(card, guiCard);
46              if (guiCard.hasText()) {
47                  cache.addText(guiCard.getText());
48              }
49          }
50      }
51      
52      /**
53       * Remove a GUICard based on a Card key.
54       * @param card The key
55       */
56      public void removeCard(Card card) {
57          if (cards.get(card).hasText()) {
58              cache.removeText(cards.get(card).getText());
59          }
60          cards.remove(card);
61          
62          int i = 0;
63          for (Card c : cards.keySet()) {
64              GUICard guiCard = cards.get(c);
65              guiCard.setPosition(x + 10 + i * (guiCard.width + 10),
66                      y + (height - guiCard.getHeight()) / 2);
67              i++;
68          }
69      }
70  }
